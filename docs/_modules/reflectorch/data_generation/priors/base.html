class PriorSampler(object):
    """Base class for prior samplers"""
    
    PARAM_CLS = Params

    @property
    def param_dim(self) -> int:
        """gets the number of parameters (i.e. the parameter dimensionality)"""
        return self.PARAM_CLS.layers_num2size(self.max_num_layers)
